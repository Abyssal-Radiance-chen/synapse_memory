def setup_milvus():
    """初始化 Milvus Collection"""
    client = MilvusClient(uri=MILVUS_URI, token=MILVUS_TOKEN)
    
    # 如果存在则删除 (方便重跑，生产环境请谨慎)
    if client.has_collection(MILVUS_COLLECTION_NAME):

        client.drop_collection(MILVUS_COLLECTION_NAME)
    
    print(f"🔨 Creating collection: {MILVUS_COLLECTION_NAME}")
    schema = MilvusClient.create_schema(auto_id=False, enable_dynamic_field=False)

def shard_files():
    """将大文件均匀拆分给每个 Worker"""
    if os.path.exists(SHARD_DIR): shutil.rmtree(SHARD_DIR)
    os.makedirs(SHARD_DIR)
    
    src_file = [f for f in os.listdir(SOURCE_FILE_DIR) if f.endswith('.jsonl')][0]
    src_path = os.path.join(SOURCE_FILE_DIR, src_file)
    
    # 打开 N 个文件句柄
    handlers = [open(os.path.join(SHARD_DIR, f"shard_{i}.jsonl"), 'w', encoding='utf-8') for i in range(len(ENDPOINTS))]
    paths = [os.path.join(SHARD_DIR, f"shard_{i}.jsonl") for i in range(len(ENDPOINTS))]
    
    print(f"🔪 Sharding file: {src_file} into {len(ENDPOINTS)} parts...")
    count = 0
    with open(src_path, 'r', encoding='utf-8') as f:
        for line in f:
            handlers[count % len(ENDPOINTS)].write(line)
            count += 1
    
    for h in handlers: h.close()
    return paths

import os
import json

from datasets import load_dataset
from tqdm.auto import tqdm

# --- 配置 ---
OUTPUT_DIR = '/kaggle/working/HotpotQA_QA_Pair'
QUESTION_FILE = os.path.join(OUTPUT_DIR, 'hotpotqa_questions.txt')
ANSWER_FILE = os.path.join(OUTPUT_DIR, 'hotpotqa_answers.txt')

if os.path.exists(OUTPUT_DIR): shutil.rmtree(OUTPUT_DIR)
os.makedirs(OUTPUT_DIR)

def generate_qa_files():
    # 1. 加载数据
    print("正在加载数据集 (hotpot_qa / distractor / validation)...")
    ds = load_dataset("hotpotqa/hotpot_qa", "distractor", split="validation")
    
    print(f"正在写入文件到 {OUTPUT_DIR} ...")
    
    with open(QUESTION_FILE, 'w', encoding='utf-8') as f_q, \
         open(ANSWER_FILE, 'w', encoding='utf-8') as f_a:
        
        for example in tqdm(ds):
            # --- 1. 处理 Question (纯文本) ---
            # 去除换行符，写入一行
            q_text = example['question'].replace('\n', ' ').strip()
            f_q.write(q_text + "\n")
            
            # --- 2. 处理 Answer (JSON 结构) ---
            
            # A. 构建 {Title: Content} 查找字典
            context_map = {}
            for title, sentences in zip(example['context']['title'], example['context']['sentences']):
                # 将句子列表拼接，并清洗换行符
                full_text = "".join(sentences).replace('\n', ' ').strip()
                context_map[title] = full_text
            
            # B. 获取并排序 supporting_facts (Evidence)
            fact_titles = example['supporting_facts']['title']
            seen_titles = set()
            evidence_list = []
            
            for t in fact_titles:
                # 去重同时保持顺序
                if t not in seen_titles:
                    # 确保 title 在 context 中存在 (极少数情况可能缺失)
                    if t in context_map:
                        evidence_list.append({
                            "title": t,
                            "content": context_map[t]
                        })
                    seen_titles.add(t)
            
            # C. 构建 JSON 对象
            # 包含简短答案和结构化的证据列表
            answer_obj = {
                "answer": example['answer'].strip(),
                "evidence": evidence_list
            }
            
            # D. 将 JSON 对象转为字符串写入一行
            f_a.write(json.dumps(answer_obj, ensure_ascii=False) + "\n")

    print("\n✅ 处理完成！")
    print(f"📂 问题文件: {QUESTION_FILE}")
    print(f"📂 答案文件: {ANSWER_FILE}")

    # --- 预览 ---
    print("\n🔍 预览 hotpot_qa_questions.txt (第一行):")
    with open(QUESTION_FILE, 'r', encoding='utf-8') as f:
        print(f.readline().strip())
        
    print("\n🔍 预览 hotpot_qa_answers.txt (第一行 - 格式化显示):")
    with open(ANSWER_FILE, 'r', encoding='utf-8') as f:
        # 读取一行（即一个 JSON 字符串）
        line = f.readline().strip()
        # 解析为 JSON 对象以便打印查看
        obj = json.loads(line)
        print(json.dumps(obj, indent=2, ensure_ascii=False))

if __name__ == "__main__":
    generate_qa_files()
